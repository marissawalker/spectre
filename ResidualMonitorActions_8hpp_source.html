90 template <typename BroadcastTarget>
91 struct UpdateResidual {
92  template <typename... DbTags, typename... InboxTags, typename Metavariables,
93  typename ArrayIndex, typename ActionList,
94  typename ParallelComponent,
95  Requires<tmpl2::flat_any_v<cpp17::is_same_v<
96  db::add_tag_prefix<LinearSolver::Tags::ResidualMagnitudeSquare,
97  typename Metavariables::system::fields_tag>,
98  DbTags>...>> = nullptr>
99  static auto apply(db::DataBox<tmpl::list<DbTags...>>& box,
100  tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
101  Parallel::ConstGlobalCache<Metavariables>& cache,
102  const ArrayIndex& /*array_index*/,
103  const ActionList /*meta*/,
104  const ParallelComponent* const /*meta*/,
105  const double res_new) noexcept {
106  using fields_tag = typename Metavariables::system::fields_tag;
107  using residual_square_tag =
108  db::add_tag_prefix<LinearSolver::Tags::ResidualMagnitudeSquare,
109  fields_tag>;
110 
111  const double residual = sqrt(res_new);
112  const double res_ratio = res_new / get<residual_square_tag>(box);
113 
114  if (static_cast<int>(get<::Tags::Verbosity>(box)) >=
115  static_cast<int>(::Verbosity::Verbose)) {
116  Parallel::printf(
117  "Linear solver iteration %d done. Remaining residual: %e\n",
118  get<LinearSolver::Tags::IterationId>(box).step_number + 1, residual);
119  }
120 
121  db::mutate<residual_square_tag, LinearSolver::Tags::IterationId>(
122  make_not_null(&box), [res_new](const gsl::not_null<double*> res_old,
123  const gsl::not_null<IterationId*>
124  iteration_id) noexcept {
125  *res_old = res_new;
126  iteration_id->step_number++;
127  });
128 
129  Parallel::simple_action<UpdateOperand>(
130  Parallel::get_parallel_component<BroadcastTarget>(cache), res_ratio,
131  equal_within_roundoff(residual, 0.));
132  }
133 };
