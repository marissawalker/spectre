18 /*!
19  * \ingroup DataStructuresGroup
20  * \brief A dynamically sized vector of arbitrary type.
21  *
22  * \details Use this vector type to represent a generic collection of numbers
23  * without assigning any particular meaning to them. It supports all common
24  * vector operations such as elementwise arithmetic, dot-products between
25  * vectors and multiplication by a DenseMatrix.
26  *
27  * \note This is a thin wrapper around `blaze::DynamicVector`. Please refer to
28  * the [Blaze documentation](https://bitbucket.org/blaze-lib/blaze/wiki/Home)
29  * for information on how to use it.
30  *
31  * Refer to the \ref DataStructuresGroup documentation for a list of other
32  * available vector and matrix types. In particular, to represent the values of
33  * a function on the computational domain use DataVector instead.
34  */
35 template <typename T, bool TF = blaze::defaultTransposeFlag>
36 class DenseVector : public blaze::DynamicVector<T, TF> {
37  public:
38  // Inherit constructors
39  using blaze::DynamicVector<T, TF>::DynamicVector;
40 
41  /// Charm++ serialization
42  // clang-tidy: runtime-references
43  void pup(PUP::er& p) noexcept { // NOLINT
44  auto size = blaze::DynamicVector<T, TF>::size();
45  p | size;
46  if (p.isUnpacking()) {
47  blaze::DynamicVector<T, TF>::resize(size);
48  }
49  if (cpp17::is_fundamental_v<T>) {
50  PUParray(p, blaze::DynamicVector<T, TF>::data(), size);
51  } else {
52  for (auto& element : *this) {
53  p | element;
54  }
55  }
56  }
57 };
