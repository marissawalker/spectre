26 /// \brief ParallelComponent representing a set of points to be interpolated
27 /// to and a function to call upon interpolation to those points.
28 ///
29 /// Each InterpolationTarget will communicate with the `Interpolator`.
30 ///
31 /// `InterpolationTargetTag` must contain the following type aliases:
32 /// - vars_to_interpolate_to_target: a `tmpl::list` of tags describing
33 /// variables to interpolate. Will be used
34 /// to construct a `Variables`.
35 /// - compute_items_on_source: a `tmpl::list` of compute items that uses
36 /// `Metavariables::interpolator_source_vars`
37 /// as input and computes the `Variables`
38 /// defined by `vars_to_interpolate_to_target`.
39 /// - compute_items_on_target: a `tmpl::list` of compute items that uses
40 /// `vars_to_interpolate_to_target` as input.
41 /// - compute_target_points: a `simple_action` of `InterpolationTarget`
42 /// that computes the target points and
43 /// sends them to `Interpolators`.
44 /// It takes a `temporal_id` as an extra
45 /// argument.
46 /// - post_interpolation_callback: a struct with a function
47 ///```
48 /// static void apply(const DataBox<DbTags>&,
49 /// const intrp::ConstGlobalCache<Metavariables>&,
50 /// const Metavariables::temporal_id&) noexcept;
51 ///```
52 /// that will be called when interpolation
53 /// is complete. `DbTags` includes everything
54 /// in `vars_to_interpolate_to_target`
55 /// and `compute_items_on_target`.
56 ///
57 /// `Metavariables` must contain the following type aliases:
58 /// - interpolator_source_vars: a `tmpl::list` of tags that define a
59 /// `Variables` sent from all `Element`s
60 /// to the local `Interpolator`.
61 /// - interpolation_target_tags: a `tmpl::list` of all
62 /// `InterpolationTargetTag`s.
63 /// - temporal_id: the type held by ::intrp::Tags::TemporalIds.
64 /// - domain_frame: The `::Frame` of the Domain.
65 template <class Metavariables, typename InterpolationTargetTag,
66  size_t VolumeDim>
67 struct InterpolationTarget {
68  using chare_type = ::Parallel::Algorithms::Singleton;
69  using metavariables = Metavariables;
70  using action_list = tmpl::list<>;
71  using initial_databox = db::compute_databox_type<
72  typename Actions::InitializeInterpolationTarget<InterpolationTargetTag>::
73  template return_tag_list<Metavariables, VolumeDim>>;
74  using options = tmpl::list<::OptionTags::DomainCreator<
75  VolumeDim, typename Metavariables::domain_frame>>;
76  using const_global_cache_tag_list = tmpl::list<>;
77 
78  static void initialize(
79  Parallel::CProxy_ConstGlobalCache<metavariables>& global_cache,
80  std::unique_ptr<
81  DomainCreator<VolumeDim, typename Metavariables::domain_frame>>
82  domain_creator) noexcept;
83  static void execute_next_phase(
84  typename metavariables::Phase /*next_phase*/,
85  const Parallel::CProxy_ConstGlobalCache<metavariables>&
86  /*global_cache*/) noexcept {}
87 };
