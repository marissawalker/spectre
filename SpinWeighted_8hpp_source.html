10 /*!
11  * \ingroup DataStructuresGroup
12  * \brief Make a spin-weighted type `T` with spin-weight `Spin`. Mathematical
13  * operators are restricted to addition, subtraction, multiplication and
14  * division, with spin-weights checked for validity.
15  *
16  * \details For a spin-weighted object, we limit operations to those valid for a
17  * pair of spin-weighted quantities - i.e. addition only makes sense when the
18  * two summands possess the same spin weight, and multiplication (or division)
19  * result in a summed (or subtracted) spin weight.
20  */
21 template <typename T, int Spin>
22 struct SpinWeighted {
23  T data;
24  using value_type = T;
25  constexpr static int spin = Spin;
26 };
