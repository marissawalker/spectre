29 /*!
30  * \brief Temporary action for observing volume and reduction data
31  *
32  * A few notes:
33  * - Observation frequency is currently hard-coded and must manually be updated.
34  * Look for `time_by_timestep_value` to update.
35  */
36 struct Observe {
37  struct ObserveNSlabs {
38  using type = size_t;
39  static constexpr OptionString help = {"Observe every Nth slab"};
40  };
41  struct ObserveAtT0 {
42  using type = bool;
43  static constexpr OptionString help = {"If true observe at t=0"};
44  };
45 
46  using const_global_cache_tags = tmpl::list<ObserveNSlabs, ObserveAtT0>;
47 
48  template <typename... DbTags, typename... InboxTags, typename Metavariables,
49  size_t Dim, typename ActionList, typename ParallelComponent>
50  static auto apply(db::DataBox<tmpl::list<DbTags...>>& box,
51  tuples::TaggedTuple<InboxTags...>& /*inboxes*/,
52  Parallel::ConstGlobalCache<Metavariables>& cache,
53  const ElementIndex<Dim>& array_index,
54  const ActionList /*meta*/,
55  const ParallelComponent* const /*meta*/) noexcept {
56  const auto& time_id = db::get<::Tags::TimeId>(box);
57  if (time_id.substep() != 0 or (time_id.slab_number() == 0 and
58  not Parallel::get<ObserveAtT0>(cache))) {
59  return std::forward_as_tuple(std::move(box));
60  }
61 
62  const auto& time = time_id.time();
63  const std::string element_name = MakeString{} << ElementId<Dim>(array_index)
64  << '/';
65  // We hard-code the writing frequency to large time values to avoid breaking
66  // the tests.
67  if (time_id.slab_number() >= 0 and time_id.time().is_at_slab_start() and
68  static_cast<size_t>(time_id.slab_number()) %
69  Parallel::get<ObserveNSlabs>(cache) ==
70  0) {
71  const auto& extents = db::get<::Tags::Mesh<Dim>>(box).extents();
72  // Retrieve the tensors and compute the solution error.
73  const auto& tilde_d = db::get<Tags::TildeD>(box);
74  const auto& tilde_tau = db::get<Tags::TildeTau>(box);
75  const auto& tilde_phi = db::get<Tags::TildePhi>(box);
76  const auto& tilde_s = db::get<Tags::TildeS<>>(box);
77  const auto& tilde_b = db::get<Tags::TildeB<>>(box);
78  const auto& rest_mass_density =
79  db::get<hydro::Tags::RestMassDensity<DataVector>>(box);
80  const auto& specific_internal_energy =
81  db::get<hydro::Tags::SpecificInternalEnergy<DataVector>>(box);
82  const auto& divergence_cleaning_field =
83  db::get<hydro::Tags::DivergenceCleaningField<DataVector>>(box);
84  const auto& pressure = db::get<hydro::Tags::Pressure<DataVector>>(box);
85  const auto& lorentz_factor =
86  db::get<hydro::Tags::LorentzFactor<DataVector>>(box);
87  const auto& specific_enthalpy =
88  db::get<hydro::Tags::SpecificEnthalpy<DataVector>>(box);
89  const auto& spatial_velocity =
90  db::get<hydro::Tags::SpatialVelocity<DataVector, Dim>>(box);
91  const auto& magnetic_field =
92  db::get<hydro::Tags::MagneticField<DataVector, Dim>>(box);
93 
94  const auto& inertial_coordinates =
95  db::get<::Tags::Coordinates<Dim, Frame::Inertial>>(box);
96 
97  // Compute the error in the solution, and generate tensor component list.
98  using PrimitiveVars =
99  typename Metavariables::analytic_variables_tags;
100  using solution_tag = OptionTags::AnalyticSolutionBase;
101  const auto exact_solution = Parallel::get<solution_tag>(cache).variables(
102  inertial_coordinates, time.value(), PrimitiveVars{});
103 
104  // Remove tensor types, only storing individual components.
105  std::vector<TensorComponent> components;
106  components.reserve(34);
107 
108  components.emplace_back(element_name + Tags::TildeD::name(),
109  tilde_d.get());
110  components.emplace_back(element_name + Tags::TildeTau::name(),
111  tilde_tau.get());
112  components.emplace_back(element_name + Tags::TildePhi::name(),
113  tilde_phi.get());
114  components.emplace_back(
115  element_name + hydro::Tags::RestMassDensity<DataVector>::name(),
116  rest_mass_density.get());
117  components.emplace_back(
118  element_name +
119  hydro::Tags::SpecificInternalEnergy<DataVector>::name(),
120  specific_internal_energy.get());
121  components.emplace_back(
122  element_name +
123  hydro::Tags::DivergenceCleaningField<DataVector>::name(),
124  divergence_cleaning_field.get());
125  components.emplace_back(
126  element_name + hydro::Tags::Pressure<DataVector>::name(),
127  pressure.get());
128  components.emplace_back(
129  element_name + hydro::Tags::LorentzFactor<DataVector>::name(),
130  lorentz_factor.get());
131  components.emplace_back(
132  element_name + hydro::Tags::SpecificEnthalpy<DataVector>::name(),
133  specific_enthalpy.get());
134 
135  using PlusSquare = funcl::Plus<funcl::Identity, funcl::Square<>>;
136  DataVector error =
137  tuples::get<hydro::Tags::RestMassDensity<DataVector>>(exact_solution)
138  .get() -
139  rest_mass_density.get();
140  const double rest_mass_density_error =
141  alg::accumulate(error, 0.0, PlusSquare{});
142  components.emplace_back(
143  element_name + "Error" +
144  hydro::Tags::RestMassDensity<DataVector>::name(),
145  error);
146  error = tuples::get<hydro::Tags::SpecificInternalEnergy<DataVector>>(
147  exact_solution)
148  .get() -
149  specific_internal_energy.get();
150  const double specific_internal_energy_error =
151  alg::accumulate(error, 0.0, PlusSquare{});
152  components.emplace_back(
153  element_name + "Error" +
154  hydro::Tags::SpecificInternalEnergy<DataVector>::name(),
155  error);
156  error =
157  tuples::get<hydro::Tags::Pressure<DataVector>>(exact_solution).get() -
158  pressure.get();
159  const double pressure_error =
160  alg::accumulate(error, 0.0, PlusSquare{});
161  components.emplace_back(
162  element_name + "Error" + hydro::Tags::Pressure<DataVector>::name(),
163  error);
164  for (size_t d = 0; d < Dim; ++d) {
165  const std::string component_suffix =
166  d == 0 ? "_x" : d == 1 ? "_y" : "_z";
167  components.emplace_back(
168  element_name + Tags::TildeS<>::name() + component_suffix,
169  tilde_s.get(d));
170  components.emplace_back(
171  element_name + Tags::TildeB<>::name() + component_suffix,
172  tilde_b.get(d));
173  components.emplace_back(
174  element_name +
175  hydro::Tags::SpatialVelocity<DataVector, Dim>::name() +
176  component_suffix,
177  spatial_velocity.get(d));
178  components.emplace_back(
179  element_name + hydro::Tags::MagneticField<DataVector, Dim>::name() +
180  component_suffix,
181  magnetic_field.get(d));
182 
183  error = tuples::get<hydro::Tags::SpatialVelocity<DataVector, Dim>>(
184  exact_solution)
185  .get(d) -
186  spatial_velocity.get(d);
187  components.emplace_back(
188  element_name + "Error" +
189  hydro::Tags::SpatialVelocity<DataVector, Dim>::name() +
190  component_suffix,
191  error);
192  error = tuples::get<hydro::Tags::MagneticField<DataVector, Dim>>(
193  exact_solution)
194  .get(d) -
195  magnetic_field.get(d);
196  components.emplace_back(
197  element_name + "Error" +
198  hydro::Tags::MagneticField<DataVector, Dim>::name() +
199  component_suffix,
200  error);
201  components.emplace_back(
202  element_name + ::Tags::Coordinates<Dim, Frame::Inertial>::name() +
203  component_suffix,
204  inertial_coordinates.get(d));
205  }
206 
207  // Send data to volume observer
208  auto& local_observer =
209  *Parallel::get_parallel_component<observers::Observer<Metavariables>>(
210  cache)
211  .ckLocalBranch();
212  Parallel::simple_action<observers::Actions::ContributeVolumeData>(
213  local_observer, observers::ObservationId(time),
214  observers::ArrayComponentId(
215  std::add_pointer_t<ParallelComponent>{nullptr},
216  Parallel::ArrayIndex<ElementIndex<Dim>>(array_index)),
217  std::move(components), extents);
218 
219  // Send data to reduction observer
220  using Redum = Parallel::ReductionDatum<double, funcl::Plus<>,
221  funcl::Sqrt<funcl::Divides<>>,
222  std::index_sequence<1>>;
223  using ReData = Parallel::ReductionData<
224  Parallel::ReductionDatum<double, funcl::AssertEqual<>>,
225  Parallel::ReductionDatum<size_t, funcl::Plus<>>, Redum, Redum, Redum>;
226  Parallel::simple_action<observers::Actions::ContributeReductionData>(
227  local_observer, observers::ObservationId(time),
228  std::vector<std::string>{
229  "Time", "NumberOfPoints", "RestMassDensityError",
230  "SpecificInternalEnergyError", "PressureError"},
231  ReData{time.value(),
232  db::get<::Tags::Mesh<Dim>>(box).number_of_grid_points(),
233  rest_mass_density_error, specific_internal_energy_error,
234  pressure_error});
235  }
236  return std::forward_as_tuple(std::move(box));
237  }
238 };
239 } // namespace Actions
240 } // namespace ValenciaDivClean
241 } // namespace grmhd
